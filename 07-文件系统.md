## 📋 概述

## 📝 文件操作实现

### 文件打开

```c
// do_sys_open() 系统调用
long do_sys_open(int dfd, const char __user *filename, int flags, umode_t mode)
{
    struct open_flags op;
    int fd = build_open_flags(flags, mode, &op);
    struct filename *tmp;
    
    if (fd)
        return fd;
    
    tmp = getname(filename);
    if (IS_ERR(tmp))
        return PTR_ERR(tmp);
    
    fd = get_unused_fd_flags(flags);
    if (fd >= 0) {
        struct file *f = do_filp_open(dfd, tmp, &op);
        if (IS_ERR(f)) {
            put_unused_fd(fd);
            fd = PTR_ERR(f);
        } else {
            fd_install(fd, f);
        }
    }
    putname(tmp);
    return fd;
}
```

### 文件读取

```c
// vfs_read() 函数
ssize_t vfs_read(struct file *file, char __user *buf, size_t count, loff_t *pos)
{
    ssize_t ret;
    
    if (!(file->f_mode & FMODE_READ))
        return -EBADF;
    if (!(file->f_mode & FMODE_CAN_READ))
        return -EINVAL;
    
    ret = rw_verify_area(READ, file, pos, count);
    if (ret)
        return ret;
    
    if (count > MAX_RW_COUNT)
        count = MAX_RW_COUNT;
    
    if (file->f_op->read)
        ret = file->f_op->read(file, buf, count, pos);
    else if (file->f_op->read_iter)
        ret = new_sync_read(file, buf, count, pos);
    else
        ret = -EINVAL;
    
    if (ret > 0) {
        add_rchar(current, ret);
        inc_syscr(current);
    }
    
    return ret;
}
```

### 练习 1: 简单 RAM 文件系统

```c
// ramfs_simple.c
#include <linux/module.h>
#include <linux/fs.h>
#include <linux/pagemap.h>
#include <linux/magic.h>

static struct dentry *root_dentry;
static struct file_system_type ramfs_fs_type;

static int ramfs_fill_super(struct super_block *sb, void *data, int silent)
{
    struct inode *inode;
    
    sb->s_maxbytes = MAX_LFS_FILESIZE;
    sb->s_blocksize = PAGE_SIZE;
    sb->s_blocksize_bits = PAGE_SHIFT;
    sb->s_magic = RAMFS_MAGIC;
    sb->s_op = &simple_super_operations;
    
    inode = new_inode(sb);
    if (!inode)
        return -ENOMEM;
    
    inode->i_mode = S_IFDIR | 0755;
    inode->i_uid = current_fsuid();
    inode->i_gid = current_fsgid();
    inode->i_atime = inode->i_mtime = inode->i_ctime = current_time(inode);
    inode->i_op = &simple_dir_inode_operations;
    inode->i_fop = &simple_dir_operations;
    inode->i_ino = 1;
    inode->i_mapping->a_ops = &ramfs_aops;
    
    root_dentry = d_make_root(inode);
    if (!root_dentry) {
        iput(inode);
        return -ENOMEM;
    }
    
    sb->s_root = root_dentry;
    return 0;
}

static struct dentry *ramfs_mount(struct file_system_type *fs_type,
                                   int flags, const char *dev_name,
                                   void *data)
{
    return mount_nodev(fs_type, flags, data, ramfs_fill_super);
}

static struct file_system_type ramfs_fs_type = {
    .owner = THIS_MODULE,
    .name = "ramfs_simple",
    .mount = ramfs_mount,
    .kill_sb = kill_litter_super,
};

static int __init ramfs_init(void)
{
    return register_filesystem(&ramfs_fs_type);
}

static void __exit ramfs_exit(void)
{
    unregister_filesystem(&ramfs_fs_type);
}

module_init(ramfs_init);
module_exit(ramfs_exit);
MODULE_LICENSE("GPL");
```

```c
// fs_monitor.c
#include <linux/module.h>
#include <linux/fsnotify.h>
#include <linux/path.h>

static struct fsnotify_group *group;

static int handle_event(struct fsnotify_group *group,
                        struct inode *to_tell,
                        struct inode *data,
                        u32 mask,
                        const void *data2,
                        size_t data2_len,
                        const unsigned char *file_name,
                        u32 cookie)
{
    pr_info("FS Event: mask=0x%x on %s\n", mask, file_name ?: "unknown");
    return 0;
}

static const struct fsnotify_ops fs_ops = {
    .handle_event = handle_event,
};

static int __init fs_monitor_init(void)
{
    group = fsnotify_alloc_group(&fs_ops);
    if (IS_ERR(group))
        return PTR_ERR(group);
    
    // 监控所有文件系统事件
    fsnotify_add_mark(group, NULL, NULL, NULL, 0,
                      FS_ALL_EVENTS);
    
    pr_info("Filesystem monitor initialized\n");
    return 0;
}

static void __exit fs_monitor_exit(void)
{
    fsnotify_destroy_group(group);
    pr_info("Filesystem monitor removed\n");
}

module_init(fs_monitor_init);
module_exit(fs_monitor_exit);
MODULE_LICENSE("GPL");
```
