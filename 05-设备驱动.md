## 📋 概述

Linux 设备驱动负责：

### 字符驱动示例

```c
// char_dev.c
#include <linux/module.h>
#include <linux/fs.h>
#include <linux/cdev.h>
#include <linux/device.h>

#define DEVICE_NAME "my_chardev"
#define CLASS_NAME "myclass"

static dev_t dev_num;
static struct cdev my_cdev;
static struct class *my_class;
static char buffer[256];
static int buffer_size = 0;

// 打开
static int char_open(struct inode *inode, struct file *file)
{
    pr_info("Device opened\n");
    return 0;
}

// 释放
static int char_release(struct inode *inode, struct file *file)
{
    pr_info("Device closed\n");
    return 0;
}

// 读
static ssize_t char_read(struct file *file, char __user *buf,
                         size_t count, loff_t *offset)
{
    int bytes_to_read = min(count, (size_t)(buffer_size - *offset));
    
    if (bytes_to_read <= 0)
        return 0;
    
    if (copy_to_user(buf, buffer + *offset, bytes_to_read))
        return -EFAULT;
    
    *offset += bytes_to_read;
    pr_info("Device read %d bytes\n", bytes_to_read);
    return bytes_to_read;
}

// 写
static ssize_t char_write(struct file *file, const char __user *buf,
                          size_t count, loff_t *offset)
{
    int bytes_to_write = min(count, sizeof(buffer));
    
    if (copy_from_user(buffer, buf, bytes_to_write))
        return -EFAULT;
    
    buffer_size = bytes_to_write;
    pr_info("Device wrote %d bytes\n", bytes_to_write);
    return bytes_to_write;
}

// 文件操作
static struct file_operations fops = {
    .owner = THIS_MODULE,
    .open = char_open,
    .release = char_release,
    .read = char_read,
    .write = char_write,
};

// 模块初始化
static int __init char_init(void)
{
    // 分配设备号
    if (alloc_chrdev_region(&dev_num, 0, 1, DEVICE_NAME) < 0)
        return -1;
    
    // 初始化 cdev
    cdev_init(&my_cdev, &fops);
    my_cdev.owner = THIS_MODULE;
    
    // 添加 cdev
    if (cdev_add(&my_cdev, dev_num, 1) < 0) {
        unregister_chrdev_region(dev_num, 1);
        return -1;
    }
    
    // 创建设备类
    my_class = class_create(CLASS_NAME);
    if (IS_ERR(my_class)) {
        cdev_del(&my_cdev);
        unregister_chrdev_region(dev_num, 1);
        return PTR_ERR(my_class);
    }
    
    // 创建设备
    device_create(my_class, NULL, dev_num, NULL, DEVICE_NAME);
    
    pr_info("Char device registered (major=%d)\n", MAJOR(dev_num));
    return 0;
}

// 模块退出
static void __exit char_exit(void)
{
    device_destroy(my_class, dev_num);
    class_destroy(my_class);
    cdev_del(&my_cdev);
    unregister_chrdev_region(dev_num, 1);
    pr_info("Char device unregistered\n");
}

module_init(char_init);
module_exit(char_exit);
MODULE_LICENSE("GPL");
MODULE_AUTHOR("Your Name");
MODULE_DESCRIPTION("Simple Character Device Driver");
```

## 🔔 中断处理

### 中断注册

```c
// 申请中断
int request_irq(unsigned int irq, irq_handler_t handler,
                unsigned long flags, const char *name, void *dev_id);

// 释放中断
void free_irq(unsigned int irq, void *dev_id);

// 中断处理函数类型
typedef irqreturn_t (*irq_handler_t)(int irq, void *dev_id);
```

### 中断标志

```c
// 中断类型
IRQF_SHARED         // 共享中断
IRQF_TRIGGER_RISING // 上升沿触发
IRQF_TRIGGER_FALLING// 下降沿触发
IRQF_TRIGGER_HIGH   // 高电平触发
IRQF_TRIGGER_LOW    // 低电平触发
IRQF_ONESHOT        // 一次性中断
IRQF_NO_THREAD      // 不线程化

// 中断返回值
IRQ_HANDLED         // 已处理
IRQ_NONE            // 未处理
```

### 中断处理示例

```c
static irqreturn_t my_irq_handler(int irq, void *dev_id)
{
    struct my_device *dev = (struct my_device *)dev_id;
    
    // 检查是否是我的中断
    if (!device_interrupt_pending(dev))
        return IRQ_NONE;
    
    // 处理中断
    disable_irq_nosync(irq);
    
    // 调度底半部
    tasklet_schedule(&dev->tasklet);
    
    return IRQ_HANDLED;
}

// 注册中断
request_irq(irq_num, my_irq_handler, 
            IRQF_TRIGGER_RISING, "my-device", dev);
```

### 底半部机制

```c
// Tasklet
DECLARE_TASKLET(my_tasklet, my_tasklet_func, data);
tasklet_schedule(&my_tasklet);

// Workqueue
DECLARE_WORK(my_work, my_work_func);
schedule_work(&my_work);

// 线程化中断
request_threaded_irq(irq, NULL, my_thread_fn,
                     IRQF_ONESHOT, "my-device", dev);
```

### 平台驱动示例

```c
static int my_platform_probe(struct platform_device *pdev)
{
    struct resource *res;
    void __iomem *base;
    int irq;
    
    // 获取内存资源
    res = platform_get_resource(pdev, IORESOURCE_MEM, 0);
    base = devm_ioremap_resource(&pdev->dev, res);
    if (IS_ERR(base))
        return PTR_ERR(base);
    
    // 获取中断
    irq = platform_get_irq(pdev, 0);
    if (irq < 0)
        return irq;
    
    // 请求中断
    devm_request_irq(&pdev->dev, irq, my_irq_handler,
                     0, "my-device", pdev);
    
    return 0;
}

static struct platform_driver my_driver = {
    .probe = my_platform_probe,
    .remove = my_platform_remove,
    .driver = {
        .name = "my-device",
        .of_match_table = my_of_match,
    },
};

module_platform_driver(my_driver);
```

## 🎯 实践练习

### 练习 1: LED 驱动

```c
// led_driver.c
#include <linux/module.h>
#include <linux/platform_device.h>
#include <linux/leds.h>
#include <linux/gpio/consumer.h>

struct my_led {
    struct led_classdev cdev;
    struct gpio_desc *gpio;
};

static void my_led_brightness_set(struct led_classdev *led,
                                  enum led_brightness brightness)
{
    struct my_led *my_led = container_of(led, struct my_led, cdev);
    gpiod_set_value(my_led->gpio, brightness);
}

static int my_led_probe(struct platform_device *pdev)
{
    struct my_led *my_led;
    
    my_led = devm_kzalloc(&pdev->dev, sizeof(*my_led), GFP_KERNEL);
    if (!my_led)
        return -ENOMEM;
    
    // 获取 GPIO
    my_led->gpio = devm_gpiod_get(&pdev->dev, "led", GPIOD_OUT_LOW);
    if (IS_ERR(my_led->gpio))
        return PTR_ERR(my_led->gpio);
    
    // 注册 LED
    my_led->cdev.name = "my-led";
    my_led->cdev.max_brightness = 1;
    my_led->cdev.brightness_set = my_led_brightness_set;
    
    return led_classdev_register(&pdev->dev, &my_led->cdev);
}

static int my_led_remove(struct platform_device *pdev)
{
    struct my_led *my_led = platform_get_drvdata(pdev);
    led_classdev_unregister(&my_led->cdev);
    return 0;
}

static const struct of_device_id my_led_of_match[] = {
    { .compatible = "vendor,my-led" },
    { }
};

static struct platform_driver my_led_driver = {
    .probe = my_led_probe,
    .remove = my_led_remove,
    .driver = {
        .name = "my-led",
        .of_match_table = my_led_of_match,
    },
};

module_platform_driver(my_led_driver);
MODULE_LICENSE("GPL");
```

### 练习 2: GPIO 中断

```c
// gpio_irq.c
#include <linux/module.h>
#include <linux/interrupt.h>
#include <linux/gpio/consumer.h>

static struct gpio_desc *button_gpio;
static int irq_num;

static irqreturn_t button_irq_handler(int irq, void *dev_id)
{
    pr_info("Button pressed! IRQ: %d\n", irq);
    return IRQ_HANDLED;
}

static int __init gpio_irq_init(void)
{
    // 获取 GPIO
    button_gpio = gpiod_get(NULL, "button", GPIOD_IN);
    if (IS_ERR(button_gpio))
        return PTR_ERR(button_gpio);
    
    // 获取中断号
    irq_num = gpiod_to_irq(button_gpio);
    if (irq_num < 0) {
        gpiod_put(button_gpio);
        return irq_num;
    }
    
    // 申请中断
    if (devm_request_irq(NULL, irq_num, button_irq_handler,
                         IRQF_TRIGGER_RISING, "button", NULL)) {
        gpiod_put(button_gpio);
        return -EINVAL;
    }
    
    pr_info("GPIO IRQ registered: %d\n", irq_num);
    return 0;
}

static void __exit gpio_irq_exit(void)
{
    devm_free_irq(NULL, irq_num, NULL);
    gpiod_put(button_gpio);
}

module_init(gpio_irq_init);
module_exit(gpio_irq_exit);
MODULE_LICENSE("GPL");
```

## ✅ 总结

Linux 设备驱动核心要点：

- **驱动模型** - device/driver/bus 架构
- **字符设备** - cdev、file_operations
- **中断处理** - request_irq、底半部
- **DMA 操作** - 一致性/流式 DMA 映射
- **设备树** - 硬件描述和匹配
- **平台驱动** - SoC 设备标准接口
